# Two Sum - Easy (#1)
# Approach: Hashmap - store seen numbers and their index
# Day 1 - 30 March 2026

class Solution:
    def twoSum(self, nums, target):
        seen = {}
        for i, num in enumerate(nums):
            diff = target - num
            if diff in seen:
                return [seen[diff], i]
            seen[num] = i
